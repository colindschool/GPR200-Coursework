//	 File name: pBlurVS
//	 Purpose: Passes the texcoordinate as NDC.
//		Also passes the inverse resolution based on the channel

layout (location = 0) in vec4 aPosition;

uniform vec2 uResolution;

out vec2 vTexcoord;
out vec2 vInvResolution;

void main()
{
	gl_Position = aPosition;
	vTexcoord = aPosition.xy;
	vInvResolution = 1.0 / uResolution;
	
}
