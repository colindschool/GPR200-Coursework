//	 File name: pDisplayFS
//	 Purpose: Sample from the blurred and the original texture.
//		Then display them with different compositions
#ifdef GL_ES
precision highp float;
#endif // GL_ES

layout (location = 0) out vec4 rtFragColor;

uniform sampler2D uTex;
uniform sampler2D uTexOriginal;

in vec2 vTexcoord;

void main()
{
	vec4 original = texture(uTexOriginal, vTexcoord);
	vec4 blur = texture(uTex, vTexcoord);
	rtFragColor = 1.0 - (1.0 - original) * (1.0 - blur);
	rtFragColor = original + blur;
	//rtFragColor = blur;
	//rtFragColor = original;
}
