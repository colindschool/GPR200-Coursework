#version 300 es

#ifdef GL_ES
precision highp float;
#endif // GL_ES

layout (location = 0) out vec4 rtFragColor;

uniform sampler2D uTex;
uniform sampler2D uTexOriginal;

in vec2 vTexcoord;

void main()
{
	vec2 uv = vTexcoord * 0.5 + 0.5;
	vec4 original = texture(uTexOriginal, uv);
	vec4 blur = texture(uTex, uv);
	rtFragColor = 1.0 - (1.0 - original) * (1.0 - blur);
	rtFragColor = original + blur;
}
